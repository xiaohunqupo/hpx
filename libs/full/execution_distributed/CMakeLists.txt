  SOURCES ${execution_distributed_sources}
  HEADERS ${execution_distributed_headers}
  DEPENDENCIES hpx_core
  MODULE_DEPENDENCIES hpx_actions_base hpx_components_base hpx_naming_base
  CMAKE_SUBDIRS examples tests
)
